def normalize_page_rank(page_rank_matrix):
    
    """Function to normalize the page rank results"""
    
    return page_rank_matrix / np.sum(page_rank_matrix)

def average_page_rank_scores(arr_of_matrices):
    """Function to average page rank scores for n iteration"""
    return np.mean(arr_of_matrices, axis=0)

def surf(node, page_rank_matrix, network, thr):
    """
    Recursively surf over pages for number of thr times, as explained in the lab manual.
    """
    global count
    if count >= thr:
        return

    page_rank_matrix[int(node)] += 1
    count += 1
    
    try: 
        out_edges = list(network.out_edges(node))[0]
    except:
        surf(node, page_rank_matrix, network, thr)
        return
        
    next_node = random.choice(out_edges)
    surf(next_node, page_rank_matrix, network, thr)

Exercise 2.13¶

With a small modification to the initial function, after reaching a dangling node we choose the next page uniformly randomly over all possible pages. Besides, at each iteration, with small probability (called the damping factor), we start surfing from a new node chosen uniformly at random.

def damped_surf(node, page_rank_matrix, network, thr):
    """Recursively surf over pages for number of thr times, as explained in the lab manual."""

    global count
    
    if count >= thr:
        return

    page_rank_matrix[int(node)] += 1
    count += 1
    
    random_node_prob = np.random.rand()
    # with probability 0.85 continue from the current page
    if random_node_prob > 0.15:
    
        try: 
            out_edges = list(network.out_edges(node))[0]
        except:
            # if reached dangling node randomly choose a new node 
            next_node = random.choice(list(network.nodes()))
            damped_surf(next_node, page_rank_matrix, network, thr)
            return
    
        next_node = random.choice(out_edges)
        damped_surf(next_node, page_rank_matrix, network, thr)
    # with prob. 0.15 jump to a random page
    else:
        next_node = random.choice(list(network.nodes()))
        damped_surf(next_node, page_rank_matrix, network, thr)

  • Random surf model makes more intuitive sense when modified with damping factor and after avoiding dangling node.

  • The probability for a random surfer be present in node is better approximated with modified random surfer for the following intuitive reasons:

    • On the web when a user is present on a web page without outgoing links, that user that user does not stay in that page forever.
    • Users do not only choose a website from the links in the pages that they are already surfing on but they can also use keywords to visit new pages.

def create_markov_matrix(H, g):
    """Populates the transition matrix of the Markov chain
    describing the random walk on the web graph
    """
    
    for i in range(g.order()):
        o_u = len(g.out_edges(str(i)))
        out_edges_of_node_i = [int(tpl[1]) for tpl in g.out_edges(str(i))]
        
        try:
            H[i, out_edges_of_node_i] = 1/o_u
        except:
            continue
    return H

def add_random_restarts(H, theta):
    """At every iteration, flip a coin, with probability  walk on H 
    and with probability 1   jump to a new random webpage.
    """
    uniform_matrix = np.ones((H.shape[0], H.shape[0])) * (1/H.shape[0])
    randomized_m = (theta*H) + (1-theta)*uniform_matrix
    return randomized_m
