# We split up our time into 100 equal width bins              

# Susceptible nodes - plot infection times with reversed direction of accumulation 
#                    ie. each bin gives the count for all bins of larger values
S, _, _ = plt.hist(infection_times, bins = 100, range=[0.0, 100.0], cumulative=-1, 
                               histtype='step', label='S', density=True, color='blue')

# Recovered nodes - plot recovery times cumulatively
#                  ie. each bin gives the count for that bin plus all bins of smaller values
R, _, _ = plt.hist(recovery_times, bins = 100, range=[0.0, 100.0], cumulative=True, 
                               histtype='step', label='R', density=True, color='green')

# Infected nodes - plot 1 - (S + R) 
#         (we cannot plot infection_times cumulatively as then we don't take into account when a node recovers)
plt.step(range(100), 1 - (S + R), where='post', label='I', color='red')

plt.title('Evolution of epidemic over time')
plt.xlabel('Time (Days)')
plt.ylabel('% Nodes that are S, I or R')

# Strategy 1
def Strategy1(budget, nb_simulations):
    G_1 = G.copy()
    for _ in range(budget):
        random_edge = random.choice(list(G_1.edges()))
        G_1.remove_edge(random_edge[0], random_edge[1])
    
    S_av = 0
    I_av = 0
    R_av = 0
    for _ in range(nb_simulations):
        SIR = epidemics_helper.SimulationSIR(G_1, beta=10.0, gamma=0.1)
        patient_0 = random.choice(list(G_1.nodes()))
    
        SIR.launch_epidemic(source = patient_0, max_time = 30)
        
        n = len(list(G_1.nodes()))
        S_av += np.sum(SIR.status == 0) / n
        I_av += np.sum(SIR.status == 1) / n
        R_av += np.sum(SIR.status == 2) / n
    
    print(f'After {nb_simulations} simulations, on average {100 * S_av / nb_simulations} % are healthy on Day 30')
    print(f'After {nb_simulations} simulations, on average {100 * I_av / nb_simulations} % are infected on Day 30')

Exercise 2.11: Strategy 2¶

The strategy we decided on is not to remove edges at random, but rather attack local bridges. In the course we learned that social ties to other communities usually go through weak links. Thus by attacking and removing local bridges we break these ties between communities and increase the shortest paths between communities to try to contain infections within the community they start in.

Furthermore in the documentation of NetworkX we saw that the function local_bridges() can also return the span of the local bridges - which "is the shortest path length between the endpoints if the local bridge is removed." Thus by sorting this set of bridges in descending order, we can remove the local bridges with the largest spans, and contain infections even more effectively.

# Strategy 2
def Strategy2(budget, nb_simulations):
    G_2 = G.copy()
    
    # span of a local bridge is the shortest path length between the endpoints if the local bridge is removed.
    local_bridges = list(nx.local_bridges(G_2, with_span=True))
    
    # sort these local bridges according to their span in descending order
    local_bridges = sorted(local_bridges, key=lambda x: - x[2])
    
    for i in range(budget):
        high_span_edge = local_bridges[i]
        G_2.remove_edge(high_span_edge[0], high_span_edge[1])
    
    S_av = 0
    I_av = 0
    R_av = 0
    for _ in range(nb_simulations):
        SIR = epidemics_helper.SimulationSIR(G_2, beta=10.0, gamma=0.1)
        patient_0 = random.choice(list(G_2.nodes()))
    
        SIR.launch_epidemic(source = patient_0, max_time = 30)
        
        n = len(list(G_2.nodes()))
        S_av += np.sum(SIR.status == 0) / n
        I_av += np.sum(SIR.status == 1) / n
        R_av += np.sum(SIR.status == 2) / n
    
    print(f'After {nb_simulations} simulations, on average {100 * S_av / nb_simulations} % are healthy on Day 30')
    print(f'After {nb_simulations} simulations, on average {100 * I_av / nb_simulations} % are infected on Day 30')
    print(f'After {nb_simulations} simulations, on average {100 * R_av / nb_simulations} % are recovered on Day 30')
