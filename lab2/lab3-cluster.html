def make_plot(dimension1, dimension2,title):
    colors = ["blue", "green", "yellow", "orange", "red"]
    
    make_name = HoverTool(tooltips=[('Name','@name'),])
    plot = figure(plot_width=960, plot_height=240, tools=[make_name, ResetTool(), PanTool(), WheelZoomTool(), SaveTool()], 
                  title=title)
    
    clusters = kmeans.cluster_centers_

    CDS = ColumnDataSource(
        data = {
            'dimension1_pos': [x[dimension1] for x in data],
            'dimension2_pos': [x[dimension2] for x in data],
            'name': list(tag2coordinates.keys()),
            'color': [colors[cluster] for cluster in kmeans.labels_]
        }
    )
    plot.diamond('dimension1_pos','dimension2_pos', source=CDS, size=3, alpha=0.5, color='color')
    
    CDS = ColumnDataSource(
        data = {
            'dimension1_pos': [x[dimension1] for x in clusters],
            'dimension2_pos': [x[dimension2] for x in clusters],
            'name': ['cluster_center' for i in range(5)]
        }
    )
    plot.triangle('dimension1_pos', 'dimension2_pos', source=CDS, size=10, alpha=0.6, color='black')

    show(plot, notebook_handle=True)

# Compute the Jaccard distance between two sets
def Jac_dist(x,y):
    return 1 - len(moviesToGenres[x].intersection(moviesToGenres[y])) / len(moviesToGenres[x].union(moviesToGenres[y]))

    
# k-medioids algorithm
def k_medioids(xs, k):
    medioids = np.random.choice(xs, k, replace = False)
    new_medioids = []
    C = []
    
    while not np.array_equal(new_medioids, medioids):
        #Actualise medioids
        if len(new_medioids) > 0:
            medioids = new_medioids
        # Initialize clusters
        C = [set() for i in range(k)]
        # Associate each point in xs to a cluster
        for x in xs:
            min_dis = 1.1
            min_index = None
            # Retrieve minimal jaccard distance between x and each one of the medioids
            for i in range(k):
                distance = Jac_dist(x, medioids[i])
                if distance < min_dis:
                    min_dis = distance
                    min_index = i
            C[min_index].add(x)

        # Compute the new medioids
        new_medioids = []

        for i in range(k):
            argmin = None
            minSum = None
            for x in C[i]:
                xSum = 0
                for y in C[i]:
                    xSum += Jac_dist(x,y)
                if minSum == None or xSum < minSum :
                    minSum = xSum
                    argmin = x
            new_medioids.append(argmin)
            
    return C

from collections import Counter

# Return dictionary of genre -> number of occurences in cluster_genres 
def nb_occurences_of_genres(cluster_genres):
    occurences = []
    for genres in cluster_genres:
        for genre in genres:
            occurences.append(genre)
    return Counter(occurences)
